# R2 (Object Storage)

R2 is Cloudflare's object storage (like S3). Use it for files, images, backups, or any binary data.

## Basic Usage

### Get a file

```gleam
import gflare/bindings
import gflare/r2
import gflare/error
import gflare/response
import gleam/javascript/promise

pub fn fetch(request, env, ctx) {
  case bindings.r2(env, "ASSETS") {
    Error(e) -> {
      response.new(500)
      |> response.set_body(error.to_string(e))
      |> promise.resolve
    }
    Ok(bucket) -> {
      // Get the file key from the URL path
      let url = request.url(request)
      let key = string.drop_start(url, 1) // Remove leading "/"

      use result <- promise.await(r2.get(bucket, key))
      case result {
        Ok(body) -> {
          // Read the body as text
          use text <- promise.await(r2.read_text(body))
          case text {
            Ok(content) -> {
              response.new(200)
              |> response.set_body(content)
              |> promise.resolve
            }
            Error(e) -> {
              response.new(500)
              |> response.set_body(error.to_string(e))
              |> promise.resolve
            }
          }
        }
        Error(_) -> {
          response.new(404)
          |> response.set_body("File not found")
          |> promise.resolve
        }
      }
    }
  }
}
```

### Upload a file

```gleam
use body <- promise.await(request.array_buffer(request))
case body {
  Ok(bytes) -> {
    let opts = r2.put_options_with(
      http_metadata: Some(r2.HttpMetadata(
        content_type: Some("text/plain"),
        content_disposition: None,
        content_encoding: None,
        cache_control: None,
        cache_expiry: None,
      )),
      custom_metadata: None,
    )
    use result <- promise.await(r2.put(bucket, "uploads/file.txt", bytes, opts))
    case result {
      Ok(obj) -> {
        response.new(200)
        |> response.set_body("Uploaded: " <> obj.key)
        |> promise.resolve
      }
      Error(e) -> {
        response.new(500)
        |> response.set_body(error.to_string(e))
        |> promise.resolve
      }
    }
  }
  Error(e) -> {
    response.new(400)
    |> response.set_body(e)
    |> promise.resolve
  }
}
```

### List files

```gleam
let opts = r2.list_options_with(
  prefix: Some("uploads/"),
  cursor: None,
  delimiter: None,
  limit: Some(100),
  include: None,
)
use result <- promise.await(r2.list(bucket, opts))
case result {
  Ok(list_result) -> {
    // list_result.objects: List(ListObject)
    // list_result.truncated: Bool
    // list_result.cursor: Option(String)
    list.each(list_result.objects, fn(obj) {
      io.println(obj.key)
    })
  }
  Error(e) -> io.println_error(error.to_string(e))
}
```

### Delete a file

```gleam
use result <- promise.await(r2.delete(bucket, "old-file.txt"))
case result {
  Ok(Nil) -> io.println("Deleted!")
  Error(e) -> io.println_error(error.to_string(e))
}
```

### Get metadata

```gleam
use result <- promise.await(r2.head(bucket, "file.txt"))
case result {
  Ok(obj) -> {
    io.println("Size: " <> int.to_string(obj.size))
  }
  Error(e) -> io.println_error(error.to_string(e))
}
```

## Reading Bodies

After getting a file body, read it in different formats:

```gleam
use body <- promise.await(r2.get(bucket, key))
case body {
  Ok(body) -> {
    // As text
    use text <- promise.await(r2.read_text(body))

    // As bytes
    use bytes <- promise.await(r2.read_bytes(body))

    // As JSON
    use json <- promise.await(r2.read_json(body))
  }
  Error(e) -> handle_error(e)
}
```

## Configuration

Add R2 bindings to `gleam.toml`:

```toml
[cloudflare.bindings]
r2 = ["ASSETS"]
```

## Options

### Put Options

```gleam
r2.put_options() // Default: no metadata

r2.put_options_with(
  http_metadata: Some(r2.HttpMetadata(
    content_type: Some("image/png"),
    content_disposition: None,
    content_encoding: None,
    cache_control: None,
    cache_expiry: None,
  )),
  custom_metadata: Some([#("author", "alice")]),
)
```

### List Options

```gleam
r2.list_options() // Default: no filters

r2.list_options_with(
  prefix: Some("images/"),
  cursor: None,
  delimiter: Some("/"),
  limit: Some(50),
  include: Some(["httpMetadata"]),
)
```

## Error Handling

```gleam
use result <- promise.await(r2.get(bucket, key))
case result {
  Ok(body) -> handle_body(body)
  Error(e) -> {
    // Common errors:
    // - R2Error("NoSuchKey") — file doesn't exist
    // - R2Error("Access Denied") — permission denied
    // - BindingNotFound("ASSETS") — binding not in gleam.toml
    io.println_error(error.to_string(e))
  }
}
```

# D1 (SQLite Database)

D1 is Cloudflare's SQLite database. Use it for structured data, user accounts, or any data that needs SQL queries.

## Basic Usage

### Query with parameters

```gleam
import gflare/bindings
import gflare/d1
import gflare/error
import gflare/response
import gleam/dynamic/decode
import gleam/javascript/promise

pub type User {
  User(id: Int, name: String, email: String)
}

// Define how to decode a User from D1 result
fn user_decoder() {
  use id <- decode.field("id", decode.int)
  use name <- decode.field("name", decode.string)
  use email <- decode.field("email", decode.string)
  decode.success(User(id:, name:, email:))
}

pub fn fetch(request, env, ctx) {
  case bindings.d1(env, "DB") {
    Error(e) -> {
      response.new(500)
      |> response.set_body(error.to_string(e))
      |> promise.resolve
    }
    Ok(db) -> {
      // Prepare and bind a query
      let stmt = d1.prepare(db, "SELECT id, name, email FROM users WHERE id = ?")
      let stmt = d1.bind(stmt, [d1.int(42)])

      // Execute and get first row
      use result <- promise.await(d1.first(stmt))
      case result {
        Ok(Some(row)) -> {
          // Decode the row into a User
          case decode.run(row, user_decoder()) {
            Ok(user) -> {
              response.new(200)
              |> response.set_body(user.name)
              |> promise.resolve
            }
            Error(_) -> {
              response.new(500)
              |> response.set_body("Failed to decode user")
              |> promise.resolve
            }
          }
        }
        Ok(None) -> {
          response.new(404)
          |> response.set_body("User not found")
          |> promise.resolve
        }
        Error(e) -> {
          response.new(500)
          |> response.set_body(error.to_string(e))
          |> promise.resolve
        }
      }
    }
  }
}
```

### Insert data

```gleam
let stmt = d1.prepare(db, "INSERT INTO users (name, email) VALUES (?, ?)")
let stmt = d1.bind(stmt, [d1.text("Alice"), d1.text("alice@example.com")])
use result <- promise.await(d1.run(stmt))
case result {
  Ok(result) -> {
    // result.meta.last_row_id contains the inserted row ID
    io.println("Created user with ID: " <> int.to_string(result.meta.last_row_id))
  }
  Error(e) -> io.println_error(error.to_string(e))
}
```

### Query all rows

```gleam
let stmt = d1.prepare(db, "SELECT id, name FROM users LIMIT 10")
use result <- promise.await(d1.all(stmt))
case result {
  Ok(result) -> {
    // result.results is List(Dynamic) — decode each row
    let users = list.map(result.results, fn(row) {
      case decode.run(row, user_decoder()) {
        Ok(user) -> user
        Error(_) -> panic as "Failed to decode user"
      }
    })
    io.println("Found " <> int.to_string(list.length(users)) <> " users")
  }
  Error(e) -> io.println_error(error.to_string(e))
}
```

### Execute raw SQL

```gleam
// For statements that don't return data
use result <- promise.await(d1.exec(db, "CREATE TABLE IF NOT EXISTS users (id INTEGER PRIMARY KEY, name TEXT)"))
case result {
  Ok(_) -> io.println("Table created!")
  Error(e) -> io.println_error(error.to_string(e))
}
```

## D1 Helper Functions

Create typed parameters with these helpers:

```gleam
d1.int(42)           // Int
d1.float(3.14)       // Float
d1.text("hello")     // String
d1.bool(True)        // Bool (stored as 0/1)
d1.blob(<<1, 2, 3>>) // BitArray
d1.null_value()      // null
```

## Configuration

Add D1 bindings to `gleam.toml`:

```toml
[cloudflare.bindings]
d1 = ["DB"]
```

## Error Handling

```gleam
use result <- promise.await(d1.first(stmt))
case result {
  Ok(Some(row)) -> handle_row(row)
  Ok(None) -> handle_not_found()
  Error(e) -> {
    // Common errors:
    // - D1Error("query failed") — SQL syntax error
    // - D1Error("no such table") — table doesn't exist
    // - BindingNotFound("DB") — binding not in gleam.toml
    io.println_error(error.to_string(e))
  }
}
```
